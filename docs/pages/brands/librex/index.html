from librex import GradientDescent

# Define your objective function
def rosenbrock(x):
    return (1 - x[0])**2 + 100*(x[1] - x[0]**2)**2

# Initialize optimizer
optimizer = GradientDescent(alpha=0.01)

# Optimize on GPU
result = optimizer.optimize(
    rosenbrock, 
    x0=[-1.0, 1.0], 
    gpu=True
)

print(f"Optimal: {result.x}")
# Optimal: [1.0, 1.0]
